一维前缀和数组

#include <iostream>
#include <vector>
using namespace std;
int main(int argc, char *argv[]) {
	vector<int> arr = {1, 2, 3, 4, 5};
	int n = arr.size();
	vector<int> sum(n, 0);
	sum[0] = arr[0];
	for(int i=1; i<n; i++){
		sum[i] = sum[i-1] + arr[i];
		cout<<sum[i]<<" ";
	}
	
	//如果要是求（i，j）之间的元素和
	//int res = sum[j]-sum[i-1]	
}

二维前缀和数组

方法一：容斥定理
for(int i=1;i<row;i++) num[i][0]=num[i-1][0]+(pizza[i][0]);
        for(int i=1;i<col;i++) num[0][i]=num[0][i-1]+(pizza[0][i]);
        for(int i=1;i<row;i++) for(int j=1;j<col;j++)
                num[i][j]=num[i-1][j]+num[i][j-1]-num[i-1][j-1]+(pizza[i][j]);
方法二：基于一维前缀和的累加
for(int i=0;i<row;i++){
    if(pizza[i][0]==1) num[i][0] = 1;
    for(int j=1;j<col;j++){
        num[i][j] = num[i][j-1] + (pizza[i][j]);
    }  
}
for(int i=1;i<row;i++){
    for(int j=0;j<col;j++){
        num[i][j] = num[i][j] + num[i-1][j];
    }  
}

但是上面求出的num数组中的每一个元素num[i][j]表示的是以左上角（0，0）右下角坐标为（i，j）的矩阵中所有元素的和。 如果要求以左上角（x,y）右下角（i，j）的矩阵中所有元素的和。则按接着如下计算：

num[i][j]-num[x-1][j]-num[i][y-1]+num[x-1][j-1]
